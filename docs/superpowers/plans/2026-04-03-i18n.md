# i18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make both GateControl clients (Pro + Community) multilingual with German + English, shared translations in client-core, live language switching.

**Architecture:** Translation engine in `@gatecontrol/client-core/src/i18n/`, shared locale JSON files for common strings, client-specific locale files in each client repo. HTML elements get `data-i18n` attributes, JS strings use `t()` calls. IPC bridge for main↔renderer locale sync.

**Tech Stack:** Vanilla JS, Electron IPC, JSON locale files, no external i18n library.

- [ ] **Step 1: Create the translation engine**

Create `/root/gatecontrol-client-core/src/i18n/index.js`:

```js
/**
 * GateControl i18n — Lightweight translation engine.
 * Shared across all GateControl clients (main + renderer process).
 */

'use strict';

let currentLocale = 'de';
const translations = {};  // { locale: { flat.key: 'value' } }
const listeners = [];

const SUPPORTED_LOCALES = ['de', 'en'];
const DEFAULT_LOCALE = 'de';

/**
 * Flatten nested object to dot-notation keys.
 * { status: { connected: 'X' } } → { 'status.connected': 'X' }
 */
function flatten(obj, prefix = '') {
  const result = {};
  for (const key of Object.keys(obj)) {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    if (typeof obj[key] === 'object' && obj[key] !== null && !Array.isArray(obj[key])) {
      Object.assign(result, flatten(obj[key], fullKey));
    } else {
      result[fullKey] = obj[key];
    }
  }
  return result;
}

/**
 * Register translations for a locale. Merges with existing.
 * @param {string} locale - 'de' or 'en'
 * @param {object} strings - Nested or flat translation object
 */
function registerTranslations(locale, strings) {
  if (!translations[locale]) translations[locale] = {};
  Object.assign(translations[locale], flatten(strings));
}

/**
 * Translate a key with optional interpolation.
 * @param {string} key - Dot-notation key, e.g. 'status.connected'
 * @param {object} [params] - Interpolation params, e.g. { days: 3 }
 * @returns {string} Translated string or key itself as fallback
 */
function t(key, params) {
  let str = translations[currentLocale]?.[key]
    || translations[DEFAULT_LOCALE]?.[key]
    || key;

  if (params) {
    for (const [k, v] of Object.entries(params)) {
      str = str.replace(new RegExp(`\\{\\{\\s*${k}\\s*\\}\\}`, 'g'), String(v));
    }
  }

  return str;
}

/**
 * Set the active locale.
 * @param {string} locale
 */
function setLocale(locale) {
  const resolved = resolveLocale(locale);
  if (resolved === currentLocale) return;
  currentLocale = resolved;
  for (const cb of listeners) {
    try { cb(currentLocale); } catch {}
  }
}

/**
 * Get the active locale.
 * @returns {string}
 */
function getLocale() {
  return currentLocale;
}

/**
 * Register a callback for locale changes.
 * @param {function} callback - Called with new locale string
 * @returns {function} Unsubscribe function
 */
function onLocaleChange(callback) {
  listeners.push(callback);
  return () => {
    const idx = listeners.indexOf(callback);
    if (idx !== -1) listeners.splice(idx, 1);
  };
}

/**
 * Map system/browser locale to supported locale.
 * 'de-DE', 'de-AT' → 'de', anything else → 'en'
 * @param {string} systemLocale
 * @returns {string}
 */
function resolveLocale(systemLocale) {
  if (!systemLocale) return DEFAULT_LOCALE;
  const lang = systemLocale.split('-')[0].toLowerCase();
  return SUPPORTED_LOCALES.includes(lang) ? lang : 'en';
}

/**
 * Get list of supported locales.
 * @returns {string[]}
 */
function getSupportedLocales() {
  return [...SUPPORTED_LOCALES];
}

module.exports = {
  t,
  setLocale,
  getLocale,
  onLocaleChange,
  registerTranslations,
  resolveLocale,
  getSupportedLocales,
};
```
